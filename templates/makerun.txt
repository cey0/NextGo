# Start Docker containers
start:
	@echo "Starting services..."
	docker-compose up -d

# Stop Docker containers
stop:
	@echo "Stopping services..."
	docker-compose down

# Rebuild and restart the Go app container
restart:
	@echo "Restarting services..."
	docker-compose up --build -d

# Initialize the Go module and tidy dependencies
init:
	@echo "Initializing Go module..."
	go mod init your_project_name || true
	go mod tidy

# Build the Go application
build:
	@echo "Building the Go app..."
	go build -o app ./cmd

# Run the Go application using Air for live reloading
run:
	@echo "Running the Go app with live reloading..."
	air

# Clean up the generated files
clean:
	@echo "Cleaning up..."
	rm -f app
