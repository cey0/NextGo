package main

import (
	"api/pkg/db"
	"api/pkg/router"
	"log"
)

func main() {
	// Initialize the database connection using the config file
	err := db.InitDB("config/config.yaml")
	if err != nil {
		log.Fatalf("Failed to initialize database: %v", err)
	}
	defer db.CloseDB()

	// Setup router
	r := router.SetupRouter()

	// Start the server
	r.Run(":8080") // or whatever port you prefer
}
