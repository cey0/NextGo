package router

import (
	"api/pkg/handlers"

	"github.com/gin-gonic/gin"
)

func SetupRouter() *gin.Engine {
	r := gin.Default()

	// Define routes
	r.GET("/health", func(c *gin.Context) {
		c.JSON(200, gin.H{
			"message": "API is up and running",
		})
	})

	r.GET("/books", handlers.GetBooks)
	r.POST("/book", handlers.CreateBook)

	return r
}
